2. Manipulasi Data dengan `dplyr`

Pustaka `dplyr` menyediakan fungsi-fungsi kuat untuk memanipulasi data frame.

# Memuat pustaka
library(dplyr)
library(ggplot2) # Untuk contoh data

# Contoh data
data("mtcars")
df_mtcars <- as_tibble(mtcars) # Mengubah ke tibble untuk dplyr

# Filter baris: pilih mobil dengan cyl = 4
filtered_data <- df_mtcars %>%
  filter(cyl == 4)
print(filtered_data)

# Pilih kolom: hanya nama model, mpg, dan hp
selected_data <- df_mtcars %>%
  select(mpg, hp, wt)
print(selected_data)

# Menambah kolom baru: rasio hp per mpg
mutated_data <- df_mtcars %>%
  mutate(hp_per_mpg = hp / mpg)
print(mutated_data)

# Mengelompokkan dan meringkas data: rata-rata mpg per jumlah silinder
summarized_data <- df_mtcars %>%
  group_by(cyl) %>%
  summarize(
    rata_rata_mpg = mean(mpg),
    count = n()
  )
print(summarized_data)

# Mengatur urutan: berdasarkan mpg secara menurun
arranged_data <- df_mtcars %>%
  arrange(desc(mpg))
print(arranged_data)

3. Visualisasi Data dengan `ggplot2`

Membuat grafik informatif dan menarik dengan `ggplot2`.

# Memuat pustaka
library(ggplot2)

# Scatter Plot: mpg vs wt
ggplot(mtcars, aes(x = wt, y = mpg)) +
  geom_point(aes(color = factor(cyl))) + # Titik dengan warna berdasarkan silinder
  labs(
    title = "Hubungan Berat Mobil dan MPG",
    x = "Berat (1000 lbs)",
    y = "Miles/(US) gallon",
    color = "Silinder"
  ) +
  theme_minimal()

# Histogram: distribusi mpg
ggplot(mtcars, aes(x = mpg)) +
  geom_histogram(binwidth = 2, fill = "steelblue", color = "black") +
  labs(
    title = "Distribusi MPG Mobil",
    x = "Miles/(US) gallon",
    y = "Frekuensi"
  ) +
  theme_light()

# Box Plot: mpg berdasarkan silinder
ggplot(mtcars, aes(x = factor(cyl), y = mpg)) +
  geom_boxplot(fill = "lightgreen") +
  labs(
    title = "Distribusi MPG berdasarkan Jumlah Silinder",
    x = "Jumlah Silinder",
    y = "Miles/(US) gallon"
  ) +
  theme_classic()

2. Manipulasi Data dengan Pandas

Pustaka `pandas` adalah tulang punggung analisis data di Python.

import pandas as pd
import numpy as np # Untuk contoh data

# Membuat DataFrame dari dictionary
data = {
    'Nama': ['Andi', 'Budi', 'Citra', 'Dini'],
    'Usia': [25, 30, 22, 28],
    'Kota': ['Jakarta', 'Bandung', 'Jakarta', 'Surabaya'],
    'Pendapatan': [5000, 6000, 4500, 7000]
}
df = pd.DataFrame(data)
print("DataFrame Awal:")
print(df)

# Memilih kolom
nama_usia = df[['Nama', 'Usia']]
print("\nNama dan Usia:")
print(nama_usia)

# Memfilter baris: Usia > 25
filtered_df = df[df['Usia'] > 25]
print("\nFilter Usia > 25:")
print(filtered_df)

# Menambah kolom baru
df['Pajak'] = df['Pendapatan'] * 0.1
print("\nDengan Kolom Pajak:")
print(df)

# Mengelompokkan dan meringkas data: rata-rata pendapatan per kota
avg_income_by_city = df.groupby('Kota')['Pendapatan'].mean().reset_index()
print("\nRata-rata Pendapatan per Kota:")
print(avg_income_by_city)

# Menggabungkan DataFrame (contoh sederhana)
df2 = pd.DataFrame({'Kota': ['Jakarta', 'Bandung'], 'Populasi': [10, 3]})
merged_df = pd.merge(df, df2, on='Kota', how='left')
print("\nDataFrame Gabungan:")
print(merged_df)

3. Visualisasi Data dengan Matplotlib & Seaborn

Membuat plot yang informatif untuk eksplorasi data.

import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd
import numpy as np

# Contoh Data
np.random.seed(42)
data = {
    'Category': np.random.choice(['A', 'B', 'C'], 100),
    'Value': np.random.normal(loc=50, scale=10, size=100),
    'Group': np.random.choice(['X', 'Y'], 100)
}
df_plot = pd.DataFrame(data)

# Scatter Plot
plt.figure(figsize=(8, 6))
sns.scatterplot(x='Value', y='Category', hue='Group', data=df_plot)
plt.title('Scatter Plot Contoh')
plt.xlabel('Nilai')
plt.ylabel('Kategori')
plt.grid(True)
plt.show()

# Histogram
plt.figure(figsize=(8, 6))
sns.histplot(df_plot['Value'], bins=10, kde=True, color='skyblue')
plt.title('Distribusi Nilai')
plt.xlabel('Nilai')
plt.ylabel('Frekuensi')
plt.show()

# Box Plot
plt.figure(figsize=(8, 6))
sns.boxplot(x='Category', y='Value', data=df_plot, palette='viridis')
plt.title('Box Plot Nilai per Kategori')
plt.xlabel('Kategori')
plt.ylabel('Nilai')
plt.show()
